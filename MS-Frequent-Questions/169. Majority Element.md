class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        dic = collections.defaultdict(int)
        for num in nums:
            dic[num] += 1
        times = 1
        res = nums[0]
        for key, value in dic.items():
            if value > times:
                res = key
                times = value
        return res
